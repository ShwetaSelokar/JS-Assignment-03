Check whether a Number is a prime or not.

            const Prime_Check = (N) => 
{
	 if(N == 1 || N == 2){
            return "NO";
          }
        var i, check=0;
        for(i=2; i< N; i++)
        {
          if(N % i === 0)
          {
            check++;
            break;
          }
        }
        if(check === 0)
          return("YES");
        else
          return("NO");
	 
	
};
